````java
```java
public int findMinArrowShots(int[][] points) {
    // 要考虑溢出问题，所以需要Integer.compare()
    Arrays.sort(points, (a, b) -> Integer.compare(a[1], b[1]));
    int arrows = 1;
    int end = points[0][1];

    for (int i = 1; i < points.length; i++) {
        if (end >= points[i][0]) {
            continue;
        }
        arrows += 1;
        end = points[i][1];
    }

    return arrows;
}
```
````
