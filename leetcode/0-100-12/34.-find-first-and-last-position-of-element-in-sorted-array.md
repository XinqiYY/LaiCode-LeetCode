## 解法一

## 代码

````java
```java
class Solution {
    public int[] searchRange(int[] a, int t) {
        int left = findLeft(a, t);
        int right = findRight(a, t);

        return new int[] { left, right };
    }

    private int findLeft(int[] a, int t) {
        if (a == null || a.length == 0) return -1;
        int left = 0;
        int right = a.length - 1;

        while (left < right - 1) {
            int mid = left + (right - left) / 2;
            if (a[mid] >= t) {
                right = mid;
            } else if (a[mid] < t) {
                left = mid;
            }
        }

        if (a[left] == t) {
            return left;
        } else if (a[right] == t) {
            return right;
        } else {
            return -1;
        }
    }

    private int findRight(int[] a, int t) {
        if (a == null || a.length == 0) return -1;
        int left = 0;
        int right = a.length - 1;

        while (left < right - 1) {
            int mid = left + (right - left) / 2;
            if (a[mid] <= t) {
                left = mid;
            } else if (a[mid] > t) {
                right = mid;
            }
        }

        if (a[right] == t) {
            return right;
        } else if (a[left] == t) {
            return left;
        } else {
            return -1;
        }
    }
}
```
````
