````java
public ListNode getIntersectionNode(ListNode headA, ListNode headB) {
    ListNode curA = headA;
    ListNode curB = headB;
    int lenA = 0, lenB = 0;
    while (curA != null) {
        lenA++;
        curA = curA.next;
    }
    while (curB != null) {
        lenB++;
        curB = curB.next;
    }

    curA = headA;
    curB = headB;
    if (lenA > lenB) {
        return helper(curA, curB, lenA, lenB);
    } else {
        return helper(curB, curA, lenB, lenA);
    }
}

private ListNode helper(ListNode curA, ListNode curB, int lenA, int lenB) {
    int gap = lenA - lenB;
    while (gap-- > 0) {
        curA = curA.next;
    }

    while (curA != null) {
        if (curA == curB) {
            return curA;
        }
        curA = curA.next;
        curB = curB.next;
    }

    return null;
}
```
````
