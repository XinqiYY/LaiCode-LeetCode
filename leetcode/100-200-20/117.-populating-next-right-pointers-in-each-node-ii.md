public Node connect(Node root) {
    if (root == null) return null;
    helper(root);
    return root;
}

private void helper(Node root) {
    Queue<Node> q = new LinkedList<>();
    q.offer(root);
    while (!q.isEmpty()) {
        int size = q.size();
        for (int i = 0; i < size; i++) {
            Node cur = q.poll();
            // 如果不check，结尾的node会链接到下一层
            if (i < size - 1) {
                cur.next = q.peek();
            }

            if (cur.left != null) {
                q.offer(cur.left);
            }

            if (cur.right != null) {
                q.offer(cur.right);
            }
        }
    }
}
```
````
